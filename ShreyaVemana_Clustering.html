3. TRAINING & TESTING THE MODEL¶

In [5]:
# Import the required libraries
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt

# Determine the optimal number of clusters using the elbow method
inertia = []
k_values = range(1, 11)
for k in k_values:
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(data_scaled)
    inertia.append(kmeans.inertia_)

# Plot the elbow curve
plt.figure(figsize=(8, 5))
plt.plot(k_values, inertia, 'bo-')
plt.xlabel('Number of clusters (k)')
plt.ylabel('Inertia')
plt.title('Elbow Method For Optimal k')
plt.show()

print("Optimal number of clusters determined using the elbow method.")

4. EVALUATION OF THE CLUSTERING MODEL¶

# Evaluate the model
# Import the required library
import seaborn as sns

# Train the KMeans model with the optimal number of clusters (e.g., 3)
optimal_k = 3
kmeans = KMeans(n_clusters=optimal_k, random_state=42)
data['Cluster'] = kmeans.fit_predict(data_scaled)

# Plot the clusters
plt.figure(figsize=(8, 5))
sns.scatterplot(data=data, x='tumor-size', y='inv-falsedes', hue='Cluster', palette='viridis')
plt.title('Cluster Visualization')
plt.xlabel('Tumor Size')
plt.ylabel('Invasive Nodes')
plt.show()

# Print cluster centroids
centroids = kmeans.cluster_centers_
centroids = scaler.inverse_transform(centroids)
centroids_df = pd.DataFrame(centroids, columns=data.columns[:-1])
print("Cluster Centroids:\n", centroids_df)

# Print number of samples in each cluster
print("Cluster Counts:\n", data['Cluster'].value_counts())
